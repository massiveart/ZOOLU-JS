 10 (function(window, ZOOLU, $, undefined) {
 11 
 12     'use strict';
 13 
 14     function log() {
 15         if (window.debug) {
 16             window.log.apply(window, $.merge(['ZOOLU', 'MIXIN'], arguments));
 17         }
 18     }
 19 
 20     /** @namespace */
 21     ZOOLU.MIXIN = {};
 22 
 23     /**
 24      * This is an event API that can be mixed into other objects.
 25      * @class
 26      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
 27      */
 28     ZOOLU.MIXIN.Events = /** @lends ZOOLU.MIXIN.Events# */ {
 29 
 30         /**
 31          * Enables event consumption and management on the provided class. This needs to be called
 32          * from the context of the object in which events are to be enabled.
 33          *
 34          * @public
 35          * @example
 36          *  var MyObj = function() {
 37          *      this.init = function() {
 38          *          ZOOLU.MIXIN.Events.enable.call(this);
 39          *      };
 40          *
 41          *      this.log = function() {
 42          *          console.log(this);
 43          *          this.trigger('log');
 44          *      };
 45          *
 46          *      this.init();
 47          *  };
 48          *
 49          *  var obj = new MyObj();
 50          *  obj.on('log', function() { console.log('Event triggered!'); });
 51          *  obj.log();
 52          */
 53         enable: function() {
 54 
 55             if (!this.listeners) {
 56                 this.listeners = { };
 57             }
 58 
 59             /** @ignore */
 60             this.trigger = function(event, args) {
 61                 ZOOLU.MIXIN.Events.trigger.call(this, event, args);
 62             }.bind(this);
 63 
 64             /** @ignore */
 65             this.on = function(event, callback) {
 66                 ZOOLU.MIXIN.Events.on.call(this, event, callback);
 67             }.bind(this);
 68 
 69             /** @ignore */
 70             this.off = function(event, callback) {
 71                 ZOOLU.MIXIN.Events.off.call(this, event, callback);
 72             }.bind(this);
 73         },
 74 
 75         /**
 76          * Triggers the provided <code>event</code> and executes all listeners attached
 77          * to it. If <code>args</code> is provided, they will be passed along to the listeners.
 78          *
 79          * @public
 80          * @param {String} event The name of the event to trigger
 81          * @param {Array} args Optional array of args to pass to the listeners
 82          */
 83         trigger: function(event, args) {
 84             if (!!this.listeners[event]) {
 85                 for (var i = -1, length = this.listeners[event].length; ++i < length;) {
 86                     // execute in the global scope (window), though this could also be customized
 87                     this.listeners[event][i].apply(window, args);
 88                 }
 89             }
 90         },
 91 
 92         /**
 93          * Binds the execution of the provided <code>callback</code> when the <code>event</code> is triggered.
 94          *
 95          * @public
 96          * @param {String} event The name of the event to bind
 97          * @param {Function} callback A function to bind to the event
 98          */
 99         on: function(event, callback) {
100 
101             if (!this.listeners[event]) {
102                 this.listeners[event] = [];
103             }
104 
105             // verify a function is being added
106             if (callback instanceof Function) {
107                 this.listeners[event].push(callback);
108             }
109         },
110 
111         /**
112          * Removes the provided <code>callback</code> from the <code>event</code>. If no function is
113          * provided, all listeners for this event are removed.
114          *
115          * @public
116          * @param {String} event The name of the event to unbind
117          * @param {Function} callback An optional listener to be removed
118          */
119         off: function(event, callback) {
120             if (!!this.listeners[event] && this.listeners[event].length > 0) {
121                 // if a listener is provided
122                 if (!!callback) {
123                     var callbacks = [];
124 
125                     for (var i = -1, length = this.listeners[event].length; ++i < length;) {
126                         if (callback !== this.listeners[event][i]) {
127                             callbacks.push(this.listeners[event][i]);
128                         }
129                     }
130                     this.listeners[event] = callbacks;
131                 } else { // no listener, so remove them all
132                     this.listeners[event] = [];
133                 }
134             }
135         }
136     };
137 
138 })(window, window.ZOOLU, window.jQuery);
