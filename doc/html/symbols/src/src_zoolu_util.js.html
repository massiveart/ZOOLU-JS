 13     function log() {
 14         if (window.debug) {
 15             window.log.apply(window, $.merge(['ZOOLU', 'UTIL'], arguments));
 16         }
 17     }

 19     /** @namespace */
 20     ZOOLU.UTIL = {};
 21 
 22     /**
 23      * This is an event API that can be mixed into other objects
 24      * @class
 25      */
 26     ZOOLU.UTIL.Events = /** @lends ZOOLU.UTIL.Events# */ {
 27 
 28         /**
 29          * Enables event consumption and management on the provided class. This needs to be called
 30          * from the context of the object in which events are to be enabled.
 31          *
 32          * @public
 33          * @example
 34          *  var MyObj = function() {
 35          *      this.init = function() {
 36          *          ZOOLU.UTIL.Events.enable.call(this);
 37          *      };
 38          *
 39          *      this.log = function() {
 40          *          console.log(this);
 41          *          this.fire('log');
 42          *      };
 43          *
 44          *      this.init();
 45          *  };
 46          *
 47          *  var obj = new MyObj();
 48          *  obj.on('log', function() { console.log('Event fired!'); });
 49          *  obj.log();
 50          */
 51         enable: function() {
 52 
 53             if (!this.listeners) {
 54                 this.listeners = { };
 55             }
 56 
 57             /** @ignore */
 58             this.fire = function(event, args) {
 59                 ZOOLU.UTIL.Events.fire.call(this, event, args);
 60             }.bind(this);
 61 
 62             /** @ignore */
 63             this.on = function(event, callback) {
 64                 ZOOLU.UTIL.Events.on.call(this, event, callback);
 65             }.bind(this);
 66 
 67             /** @ignore */
 68             this.off = function(event, callback) {
 69                 ZOOLU.UTIL.Events.off.call(this, event, callback);
 70             }.bind(this);
 71         },
 72 
 73         /**
 74          * Fires the provided <code>event</code> and executes all listeners attached
 75          * to it. If <code>args</code> is provided, they will be passed along to the listeners.
 76          *
 77          * @public
 78          * @param {String} event The name of the event to fire
 79          * @param {Array} args Optional array of args to pass to the listeners
 80          */
 81         fire: function(event, args) {
 82             if (!!this.listeners[event]) {
 83                 for (var i = -1, length = this.listeners[event].length; ++i < length;) {
 84                     // execute in the global scope (window), though this could also be customized
 85                     this.listeners[event][i].apply(window, args);
 86                 }
 87             }
 88         },
 89 
 90         /**
 91          * Binds the execution of the provided <code>callback</code> when the <code>event</code> is fired.
 92          *
 93          * @public
 94          * @param {String} event The name of the event to bind
 95          * @param {Function} callback A function to bind to the event
 96          */
 97         on: function(event, callback) {
 98 
 99             if (!this.listeners[event]) {
100                 this.listeners[event] = [];
101             }
102 
103             // verify a function is being added
104             if (callback instanceof Function) {
105                 this.listeners[event].push(callback);
106             }
107         },
108 
109         /**
110          * Removes the provided <code>callback</code> from the <code>event</code>. If no function is
111          * provided, all listeners for this event are removed.
112          *
113          * @public
114          * @param {String} event The name of the event to unbind
115          * @param {Function} callback An optional listener to be removed
116          */
117         off: function(event, callback) {
118             if (!!this.listeners[event] && this.listeners[event].length > 0) {
119                 // if a listener is provided
120                 if (!!callback) {
121                     var callbacks = [];
122 
123                     for (var i = -1, length = this.listeners[event].length; ++i < length;) {
124                         if (callback !== this.listeners[event][i]) {
125                             callbacks.push(this.listeners[event][i]);
126                         }
127                     }
128                     this.listeners[event] = callbacks;
129                 } else { // no listener, so remove them all
130                     this.listeners[event] = [];
131                 }
132             }
133         }
134     };

138      */
