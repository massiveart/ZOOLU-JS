 10 (function(window, ZOOLU, $, undefined) {
 11 
 12     'use strict';
 13 
 14     function log() {
 15         if (window.debug) {
 16             window.log.apply(window, $.merge(['ZOOLU', 'UTIL'], arguments));
 17         }
 18     }
 19 
 20     /** @namespace */
 21     ZOOLU.UTIL = {};
 22 
 23     /**
 24      * @namespace
 25      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
 26      */
 27     ZOOLU.UTIL.String = /** @lends ZOOLU.UTIL.String */ {
 28 
 29         /**
 30          * Truncate a string to the given length and adding an ellipsis.
 31          *
 32          * @static
 33          * @param {String} str String to be truncated
 34          * @param {Integer} limit Max length of the string
 35          * @return string
 36          */
 37         truncate: function(str, limit) {
 38             log('String', 'truncate');
 39 
 40             if (typeof str !== 'string') {
 41                 return '';
 42             }
 43 
 44             if (str.length > limit) {
 45                 return str.slice(0, limit - 3) + '...';
 46             } else {
 47                 return str;
 48             }
 49         },
 50 
 51         /**
 52          * Truncate a string to the given length, breaking at word boundaries and adding an ellipsis.
 53          *
 54          * @static
 55          * @param {String} str String to be truncated
 56          * @param {Integer} limit Max length of the string
 57          * @return string
 58          */
 59         truncateAtWordBoundaries: function(str, limit) {
 60             log('String', 'truncateAtWordBoundaries');
 61 
 62             var bits, i;
 63 
 64             if (typeof str !== 'string') {
 65                 return '';
 66             }
 67 
 68             bits = str.split('');
 69 
 70             if (bits.length > limit) {
 71                 for (i = bits.length - 1; i > -1; --i) {
 72                     if (i >= limit) {
 73                         bits.length = i;
 74                     } else if (' ' === bits[i]) {
 75                         bits.length = i;
 76                         break;
 77                     }
 78                 }
 79                 bits.push(' ...');
 80             }
 81 
 82             return bits.join('');
 83         },
 84 
 85         /**
 86          * Truncate a string to the given length and adding an ellipsis in between.
 87          *
 88          * @static
 89          * @param {String} str String to be truncated
 90          * @param {Integer} limit Max length of the string
 91          * @param {String} separator Separator will be added, if given string is too long
 92          * @return string
 93          */
 94         truncateInBetween: function(str, limit, separator) {
 95             log('String', 'truncateAtTheMiddle');
 96 
 97             if (typeof str !== 'string') {
 98                 return '';
 99             }
100 
101             if (str.length > limit) {
102                 separator = separator || ' ... ';
103 
104                 var sepLen = separator.length,
105                     charsToShow = limit - sepLen,
106                     frontChars = Math.ceil(charsToShow / 3 * 2),
107                     backChars = Math.floor(charsToShow / 3 * 1);
108 
109                 return str.substr(0, frontChars) + separator + str.substr(str.length - backChars);
110             } else {
111                 return str;
112             }
113         }
114     };
