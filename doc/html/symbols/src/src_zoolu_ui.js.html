 25      *
 26      * @class
 27      * @constructor
 28      * @example
 29      *  throw new ZOOLU.UI.Exception('Message for the catcher.');
 30      *
 31      * @param {String|Null} message Message which is throw to the catcher
 32      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
 33      */
 34     ZOOLU.UI.Exception = function(message) {
 35         this.name = 'ZOOLU.UI.Exception';
 36         this.message = message || 'Something went wrong!';
 37     };

 53     /**
 54      * ZOOLU ColumnTree UI Element
 55      *
 56      * @class
 57      * @constructor
 58      * @example
 59      *  var columnTree = new ZOOLU.UI.ColumnTree('#tree', {
 60      *      url: '/nodes',
 61      *      hasChildren: { 'folder': true }
 62      *  });
 63      *
 64      * @borrows ZOOLU.MIXIN.Events#trigger as #trigger
 65      * @borrows ZOOLU.MIXIN.Events#on as #on
 66      * @borrows ZOOLU.MIXIN.Events#off as #off
 67      * @param {String} container Selector to get the column tree container
 68      * @param {Object} options Default options will be merged with the given options
 69      * @throws {ZOOLU.UI.Exception} ColumnTree container doesn't exist!
 70      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
 71      */
 72     ZOOLU.UI.ColumnTree = function(container, options) {
 73         this.$container = $(container);
 74 
 75         if (!this.$container.length) {
 76             throw new ZOOLU.UI.Exception("ColumnTree container doesn't exist!");
 77         }
 78 
 79         // add css class
 80         this.$container.addClass('column-tree');
 81 
 82         // extend default options with given
 83         this.options = $.extend({
 84             hasChildren: { },
 85             urlAddOnForLoadingNodeChildren: '/children',
 86             rowMaxLength: 30
 87         }, options);
 88 
 89         this.level = 0;
 90         this.selectedId = null;
 91 
 92         this.$selected = null;
 93         this.$currentColumn = null;
 94 
 95         this.data = [];
 96 
 97         // add event API
 98         ZOOLU.MIXIN.Events.enable.call(this);
 99 
100         log('ColumnTree', 'construct', this);
101 
102         // try to load first level
103         this.load();
104     };

106     ZOOLU.UI.ColumnTree.prototype = /** @lends ZOOLU.UI.ColumnTree# */ {
107 
108         constructor: ZOOLU.UI.ColumnTree,
109 
110         /**
111          * Loads nodes from the given REST service entry point (<code>options.url</code>).
112          * If there is a selected node, the children of this node will be loaded.
113          * @triggers ColumnTree.load
114          */
115         load: function() {
116             log('ColumnTree', 'load');
117 
118             if (this.options.url) {
119                 this.trigger('ColumnTree.load');
120 
121                 // add column if is missing for the current level
122                 if (this.$container.find('.column').length <= this.level) {
123                     this.addColumn();
124                 }
125 
126                 this.$currentColumn = $('#column-' + this.level);
127                 this.$currentColumn.addClass('busy');
128 
129                 $.ajax({
130                     url: this.selectedId ? this.options.url + '/' + this.selectedId + this.options.urlAddOnForLoadingNodeChildren : this.options.url,
131                     type: 'GET',
132                     dataType: 'JSON',
133                     success: function(data) {
134                         log('ColumnTree', 'load', 'success', data);
135                         this.updateData(data);
136                         this.updateView();
137                     }.bind(this),
138                     error: function(jqXHR, textStatus) {
139                         log('ColumnTree', 'load', 'error', textStatus);
140                         this.$currentColumn.removeClass('busy');
141                         // TODO
142                     }.bind(this)
143                 });
144             }
145         },
146 
147         /**
148          *
149          */
150         updateData: function(data) {
151             log('ColumnTree', 'updateData');
152 
153             this.data[this.level] = data;
154         },
155 
156         /**
157          *
158          * @throws {ZOOLU.UI.Exception} Current column not found!
159          */
160         updateView: function() {
161             log('ColumnTree', 'updateView');
162 
163             if (!this.$currentColumn.length) {
164                 throw new ZOOLU.UI.Exception('Current column not found!');
165             }
166 
167             this.$currentColumn.removeClass('busy');
168 
169             this.$list = $('<ul class="list"/>');
170 
171             // http://jsperf.com/loops3/2
172             for (var i = -1, length = this.data[this.level].length; ++i < length;) {
173                 this.prepareRow(this.data[this.level][i]).appendTo(this.$list);
174             }
175 
176             this.$currentColumn.html(this.$list);
177 
178             // cleanup
179             delete this.$row;
180             delete this.$list;
181         },
182 
183         /**
184          *
185          */
186         addColumn: function() {
187             log('ColumnTree', 'addColumn');
188             this.$container.append($('<div class="column"/>').attr('id', 'column-' + this.level).data('level', this.level));
189         },
190 
191         /**
192          *
193          * @return {Object}
194          */
195         prepareRow: function(node) {
196             log('ColumnTree', 'prepareRow', arguments);
197 
198             this.$row = $('<li class="row ' + node.type + '"/>').attr('id', 'row-' + node.id + '-' + node.type);
199             this.$row.html(ZOOLU.UTIL.String.truncateInBetween(node.name, this.options.rowMaxLength));
200 
201             // store node data to the element
202             this.$row.data('id', node.id);
203             this.$row.data('type', node.type);
204             this.$row.data('level', this.level);
205 
206             this.attachRowObservers(node);
207 
208             return this.$row;
209         },
210 
211         /**
212          *
213          */
214         attachRowObservers: function() {
215             log('ColumnTree', 'attachRowObservers', arguments);
216 
217             this.$row.on('click', function(event) {
218                 this.select(event.target);
219             }.bind(this));
220         },
221 
222         /**
223          *
224          * @param {Object} element Element which has been selected
225          * @triggers ColumnTree.select with the selected object
226          */
227         select: function(element) {
228             log('ColumnTree', 'select', arguments);
229 
230             this.$selected = $(element);
231             this.trigger('ColumnTree.select', [this.$selected]);
232             this.updateSelectedMarker();
233 
234             this.$selected.addClass('selected');
235 
236             this.selectedId = this.$selected.data('id');
237             this.level = this.$selected.data('level');
238 
239             // cleanup tree
240             this.$container.find('.column').each(function(index, element) {
241                 if ($(element).data('level') > this.level) {
242                     $(element).html('');
243                 }
244             }.bind(this));
245 
246             if (this.options.hasChildren.hasOwnProperty(this.$selected.data('type')) &&
247                 this.options.hasChildren[this.$selected.data('type')] === true) {
248 
249                 this.level++;
250                 this.load();
251             }
252         },
253 
254         /**
255          *
256          */
257         updateSelectedMarker: function() {
258             // TODO
259         }
260     };

262     /**
263      * ZOOLU Layout UI Element
264      *
265      * @class
266      * @constructor
267      * @example
268      *  var layout = new ZOOLU.UI.Layout('#layout');
269      *
270      * @borrows ZOOLU.MIXIN.Events#trigger as #trigger
271      * @borrows ZOOLU.MIXIN.Events#on as #on
272      * @borrows ZOOLU.MIXIN.Events#off as #off
273      * @param {String} container Selector to get the layout container
274      * @param {Object} options Default options will be merged with the given options
275      * @throws {ZOOLU.UI.Exception} Layout container doesn't exist!
276      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
277      */
278     ZOOLU.UI.Layout = function(container, options) {
279         this.$container = $(container);
280 
281         if (!this.$container.length) {
282             throw new ZOOLU.UI.Exception("Layout container doesn't exist!");
283         }
284 
285         // add css class
286         this.$container.addClass('layout');
287         this.$container.addClass('busy');
288 
289         this.height = this.$container.outerHeight(true);
290         this.width = this.$container.outerWidth(true);
291 
292         this.panels = {};
293 
294         // extend default options with given
295         this.options = $.extend({
296             fullScreen: true
297         }, options);
298 
299         // add event API
300         ZOOLU.MIXIN.Events.enable.call(this);
301 
302         log('Layout', 'construct', this);
303 
304         this.initialize();
305     };

307     ZOOLU.UI.Layout.prototype = /** @lends ZOOLU.UI.Layout# */ {
308 
309         constructor: ZOOLU.UI.Layout,
310 
311         CONST: {
312             orientations: ['west', 'north', 'center']
313         },
314 
315         initialize: function() {
316             var panels;
317 
318             log('Layout', 'initialize');
319 
320             // if full screen, set some default html & body css properties
321             if (this.options.fullScreen) {
322                 $('html').css({
323                     height: '100%',
324                     overflow: 'hidden',
325                     overflowX: 'hidden',
326                     overflowY: 'hidden'
327                 });
328 
329                 $('body').css({
330                     position: 'relative',
331                     height: '100%',
332                     overflow: 'hidden',
333                     overflowX: 'hidden',
334                     overflowY: 'hidden',
335                     margin: 0,
336                     padding: 0,
337                     border: 'none'
338                 });
339             }
340 
341             // find and init all panels
342             for (var i = -1, length = this.CONST.orientations.length; ++i < length;) {
343                 panels = this.$container.find('> .' + this.CONST.orientations[i]);
344                 if (panels.length) {
345                     this.initPanel(panels.first(), this.CONST.orientations[i]);
346                 }
347             }
348 
349             this.arrange();
350 
351             // resize observer
352             $(window).resize(this.arrange.bind(this));
353         },
354 
355         enable: function() {
356 
357             /** @ignore */
358             this.initPanel = function(panel, orientation) {
359                 ZOOLU.UI.Layout.prototype.initPanel.call(this, panel, orientation);
360             }.bind(this);
361 
362             /** @ignore */
363             this.addPanel = function(panel) {
364                 ZOOLU.UI.Layout.prototype.addPanel.call(this, panel);
365             }.bind(this);
366 
367             /** @ignore */
368             this.arrange = function() {
369                 ZOOLU.UI.Layout.prototype.arrange.call(this);
370             }.bind(this);
371         },
372 
373         /**
374          * Initialize panel of the current layout.
375          *
376          * @param {Object} panel Panel of the current layout
377          * @param {String} orientation Orientation of the given panel
378          */
379         initPanel: function(panel, orientation) {
380 
381             log('Layout', 'initPanel', arguments);
382 
383             this.addPanel(new ZOOLU.UI.Layout.Panel(panel, this, orientation, this.options));
384         },
385 
386         /**
387          * Add instance of ZOOLU.UI.Layout.Panel to the current layout.
388          *
389          * @param {ZOOLU.UI.Layout.Panel} panel Instance of ZOOLU.UI.Layout.Panel
390          */
391         addPanel: function(panel) {
392 
393             log('Layout', 'addPanel', arguments);
394 
395             this.panels[panel.orientation] = panel;
396 
397             panel.on('Layout.Panel.resize', this.arrange.bind(this));
398         },
399 
400         /**
401          * Arrange all layout panels.
402          */
403         arrange: function() {
404 
405             if (!!this.layout) {
406                 log('Layout', 'Panel', 'arrange');
407             } else {
408                 log('Layout', 'arrange');
409             }
410 
411             //TODO cleanup!!!
412 
413             this.updateDimension();
414 
415             // west
416             this.left = 0;
417             if (this.panels.hasOwnProperty('west')) {
418                 this.panels.west.setPositionLeft(this.left);
419                 this.left += this.panels.west.width;
420             }
421 
422             // north
423             this.top = 0;
424             if (this.panels.hasOwnProperty('north')) {
425                 this.panels.north.setPositionTop(this.top);
426                 this.top += this.panels.north.height;
427             }
428 
429             // center
430             if (this.panels.hasOwnProperty('center')) {
431                 this.panels.center.setPositionLeft(this.left);
432                 this.panels.center.setPositionTop(this.top);
433                 this.panels.center.updateDimension(
434                     (this.left === 0 ? '100%' : this.width - this.left),
435                     (this.top === 0 ? '100%' : this.height - this.top)
436                 );
437 
438                 if (this.panels.center.hasChildren) {
439                     this.panels.center.arrange();
440                 }
441             }
442 
443             this.unbusy();
444         },
445 
446         unbusy: function() {
447             this.$container.removeClass('busy');
448         },
449 
450         updateDimension: function(width, height) {
451             if (!!width) {
452                 this.$container.width(width);
453             }
454 
455             if (!!height) {
456                 this.$container.height(height);
457             }
458 
459             this.height = this.$container.outerHeight(true);
460             this.width = this.$container.outerWidth(true);
461         }
462     };

464     /**
465      * ZOOLU Layout Panel UI Element
466      *
467      * @class
468      * @constructor
469      * @private
470      * @borrows ZOOLU.MIXIN.Events#trigger as #trigger
471      * @borrows ZOOLU.MIXIN.Events#on as #on
472      * @borrows ZOOLU.MIXIN.Events#off as #off
473      * @param {String} element Selector to get the panel
474      * @param {ZOOLU.UI.Layout} layout Instance of the parent ZOOLU.UI.Layout
475      * @param {Object} options Default options will be merged with the given options
476      * @throws {ZOOLU.UI.Exception} Panel doesn't exist!
477      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
478      */
479     ZOOLU.UI.Layout.Panel = function(element, layout, orientation, options) {
480         this.$element = $(element);
481         this.layout = layout;
482         this.orientation = orientation;
483         this.hasChildren = false;
484         this.handler = null;
485 
486         if (!this.$element.length) {
487             throw new ZOOLU.UI.Exception("Panel doesn't exist!");
488         }
489 
490         // add css class
491         this.$element.addClass('panel');
492 
493         this.height = this.$element.outerHeight(true);
494         this.width = this.$element.outerWidth(true);
495 
496         // extend default options with given
497         this.options = $.extend({
498 
499         }, options);
500 
501         // add event API
502         ZOOLU.MIXIN.Events.enable.call(this);
503 
504         log('Layout', 'Panel', 'construct', this);
505 
506         this.initialize();
507 
508         if (this.orientation === 'center') {
509             ZOOLU.UI.Layout.prototype.enable.call(this);
510             this.panels = {};
511             this.initializeLayout();
512         }
513     };

515     ZOOLU.UI.Layout.Panel.prototype = /** @lends ZOOLU.UI.Layout.Panel# */ {
516 
517         constructor: ZOOLU.UI.Layout.Panel,
518 
519         initialize: function() {
520             this.$handler = $('<div class="handler"/>');
521             this.$handler.appendTo(this.$element);
522 
523             // TODO cleanup!!!
524             this.$handler.mousedown(function(event) {
525                 event.preventDefault();
526                 $(window).on('mousemove.layout', this.resize.bind(this));
527             }.bind(this));
528 
529             $(window).mouseup(function() {
530                 $(window).off('mousemove.layout');
531             });
532         },
533 
534         initializeLayout: function() {
535             var panels;
536 
537             log('Layout', 'Panel', 'initialize');
538 
539             // find and init all panels
540             for (var i = -1, length = ZOOLU.UI.Layout.prototype.CONST.orientations.length; ++i < length;) {
541                 panels = this.$element.find('> .' + ZOOLU.UI.Layout.prototype.CONST.orientations[i]);
542                 if (panels.length) {
543                     this.hasChildren = true;
544                     this.initPanel(panels.first(), ZOOLU.UI.Layout.prototype.CONST.orientations[i]);
545                 }
546             }
547         },
548 
549         /**
550          *
551          * @param {Object} event
552          * @triggers Layout.Panel.resize
553          */
554         resize: function(event) {
555             //log(event.pageX + ', ' + event.pageY);
556 
557             if (this.orientation === 'west') {
558                 this.updateDimension(event.pageX);
559             } else {
560                 this.updateDimension(null, event.pageY);
561             }
562 
563             this.trigger('Layout.Panel.resize');
564         },
565 
566         unbusy: function() {
567             this.$element.removeClass('busy');
568         },
569 
570         updateDimension: function(width, height) {
571             if (!!width) {
572                 this.$element.width(width);
573             }
574 
575             if (!!height) {
576                 this.$element.height(height);
577             }
578 
579             this.height = this.$element.outerHeight(true);
580             this.width = this.$element.outerWidth(true);
581         },
582 
583         setPositionLeft: function(left) {
584 
585             log('Layout', 'Panel', 'setPositionLeft', this.orientation);
586 
587             this.$element.css({left: left});
588         },
589 
590         setPositionTop: function(top) {
591 
592             log('Layout', 'Panel', 'setPositionTop', this.orientation);
593 
594             this.$element.css({top: top});
595         }
596     };
