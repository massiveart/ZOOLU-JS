 53     /**
 54      * ZOOLU ColumnTree UI Element
 55      *
 56      * @class
 57      * @constructor
 58      * @example
 59      *  var columnTree = new ZOOLU.UI.ColumnTree('#tree', {
 60      *      url: '/nodes',
 61      *      hasChildren: { 'folder': true }
 62      *  });
 63      *
 64      * @borrows ZOOLU.MIXIN.Events#trigger as #trigger
 65      * @borrows ZOOLU.MIXIN.Events#on as #on
 66      * @borrows ZOOLU.MIXIN.Events#off as #off
 67      * @param {String} container Selector to get the column tree container
 68      * @param {Object} options Default options will be merged with the given options
 69      * @throws {ZOOLU.UI.Exception} ColumnTree container doesn't exist!
 70      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
 71      */
 72     ZOOLU.UI.ColumnTree = function(container, options) {
 73         this.$container = $(container);
 74 
 75         if (!this.$container.length) {
 76             throw new ZOOLU.UI.Exception("ColumnTree container doesn't exist!");
 77         }
 78 
 79         // add css class
 80         this.$container.addClass('column-tree');
 81 
 82         // extend default options with given
 83         this.options = $.extend({
 84             hasChildren: { },
 85             urlAddOnForLoadingNodeChildren: '/children',
 86             rowMaxLength: 30
 87         }, options);
 88 
 89         this.level = 0;
 90         this.selectedId = null;
 91 
 92         this.$selected = null;
 93         this.$currentColumn = null;
 94 
 95         this.data = [];
 96 
 97         // add event API
 98         ZOOLU.MIXIN.Events.enable.call(this);
 99 
100         log('ColumnTree', 'construct', this);
101 
102         // try to load first level
103         this.load();
104     };
105 
106     ZOOLU.UI.ColumnTree.prototype = /** @lends ZOOLU.UI.ColumnTree# */ {
107 
108         constructor: ZOOLU.UI.ColumnTree,
109 
110         /**
111          * Loads nodes from the given REST service entry point (<code>options.url</code>).
112          * If there is a selected node, the children of this node will be loaded.
113          * @triggers ColumnTree.load
114          */
115         load: function() {
116             log('ColumnTree', 'load');
117 
118             if (this.options.url) {
119                 this.trigger('ColumnTree.load');
120 
121                 // add column if is missing for the current level
122                 if (this.$container.find('.column').length <= this.level) {
123                     this.addColumn();
124                 }
125 
126                 this.$currentColumn = $('#column-' + this.level);
127                 this.$currentColumn.addClass('busy');
128 
129                 $.ajax({
130                     url: this.selectedId ? this.options.url + '/' + this.selectedId + this.options.urlAddOnForLoadingNodeChildren : this.options.url,
131                     type: 'GET',
132                     dataType: 'JSON',
133                     success: function(data) {
134                         log('ColumnTree', 'load', 'success', data);
135                         this.updateData(data);
136                         this.updateView();
137                     }.bind(this),
138                     error: function(jqXHR, textStatus) {
139                         log('ColumnTree', 'load', 'error', textStatus);
140                         this.$currentColumn.removeClass('busy');
141                         // TODO
142                     }.bind(this)
143                 });
144             }
145         },
146 
147         /**
148          *
149          */
150         updateData: function(data) {
151             log('ColumnTree', 'updateData');
152 
153             this.data[this.level] = data;
154         },
155 
156         /**
157          *
158          * @throws {ZOOLU.UI.Exception} Current column not found!
159          */
160         updateView: function() {
161             log('ColumnTree', 'updateView');
162 
163             if (!this.$currentColumn.length) {
164                 throw new ZOOLU.UI.Exception('Current column not found!');
165             }
166 
167             this.$currentColumn.removeClass('busy');
168 
169             this.$list = $('<ul class="list"/>');
170 
171             // http://jsperf.com/loops3/2
172             for (var i = -1, length = this.data[this.level].length; ++i < length;) {
173                 this.prepareRow(this.data[this.level][i]).appendTo(this.$list);
174             }
175 
176             this.$currentColumn.html(this.$list);
177 
178             // cleanup
179             delete this.$row;
180             delete this.$list;
181         },
182 
183         /**
184          *
185          */
186         addColumn: function() {
187             log('ColumnTree', 'addColumn');
188             this.$container.append($('<div class="column"/>').attr('id', 'column-' + this.level).data('level', this.level));
189         },
190 
191         /**
192          *
193          * @return {Object}
194          */
195         prepareRow: function(node) {
196             log('ColumnTree', 'prepareRow', arguments);
197 
198             this.$row = $('<li class="row ' + node.type + '"/>').attr('id', 'row-' + node.id + '-' + node.type);
199             this.$row.html(ZOOLU.UTIL.String.truncateInBetween(node.name, this.options.rowMaxLength));
200 
201             // store node data to the element
202             this.$row.data('id', node.id);
203             this.$row.data('type', node.type);
204             this.$row.data('level', this.level);
205 
206             this.attachRowObservers(node);
207 
208             return this.$row;
209         },
210 
211         /**
212          *
213          */
214         attachRowObservers: function() {
215             log('ColumnTree', 'attachRowObservers', arguments);
216 
217             this.$row.on('click', function(event) {
218                 this.select(event.target);
219             }.bind(this));
220         },
221 
222         /**
223          *
224          * @param {Object} element Element which has been selected
225          * @triggers ColumnTree.select with the selected object
226          */
227         select: function(element) {
228             log('ColumnTree', 'select', arguments);
229 
230             this.$selected = $(element);
231             this.trigger('ColumnTree.select', [this.$selected]);
232             this.updateSelectedMarker();
233 
234             this.$selected.addClass('selected');
235 
236             this.selectedId = this.$selected.data('id');
237             this.level = this.$selected.data('level');
238 
239             // cleanup tree
240             this.$container.find('.column').each(function(index, element) {
241                 if ($(element).data('level') > this.level) {
242                     $(element).html('');
243                 }
244             }.bind(this));
245 
246             if (this.options.hasChildren.hasOwnProperty(this.$selected.data('type')) &&
247                 this.options.hasChildren[this.$selected.data('type')] === true) {
248 
249                 this.level++;
250                 this.load();
251             }
252         },
253 
254         /**
255          *
256          */
257         updateSelectedMarker: function() {
258             // TODO
259         }
260     };

262     /**
263      * ZOOLU Layout UI Element
264      *
265      * @class
266      * @constructor
267      * @example
268      *  var layout = new ZOOLU.UI.Layout('#layout');
269      *
270      * @borrows ZOOLU.MIXIN.Events#trigger as #trigger
271      * @borrows ZOOLU.MIXIN.Events#on as #on
272      * @borrows ZOOLU.MIXIN.Events#off as #off
273      * @param {String} container Selector to get the layout container
274      * @param {Object} options Default options will be merged with the given options
275      * @throws {ZOOLU.UI.Exception} Layout container doesn't exist!
276      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
277      */
278     ZOOLU.UI.Layout = function(container, options) {
279         this.$container = $(container);
280 
281         if (!this.$container.length) {
282             throw new ZOOLU.UI.Exception("Layout container doesn't exist!");
283         }
284 
285         // add css class
286         this.$container.addClass('layout');
287 
288         this.panels = [];
289 
290         // extend default options with given
291         this.options = $.extend({
292             fullPage: true,
293             possiblePanels: ['west', 'north', 'center'],
294             hasChildPanels: ['center']
295         }, options);
296 
297         // add event API
298         ZOOLU.MIXIN.Events.enable.call(this);
299 
300         log('Layout', 'construct', this);
301 
302         this.initialize();
303     };
304 
305     ZOOLU.UI.Layout.prototype = /** @lends ZOOLU.UI.Layout# */ {
306 
307         constructor: ZOOLU.UI.Layout,
308 
309         initialize: function() {
310 
311             log('Layout', 'initialize');
312 
313             for (var i = -1, length = this.options.possiblePanels.length; ++i < length;) {
314                 this.$container.find('> .' + this.options.possiblePanels[i]).each(this.initPanel.bind(this));
315             }
316         },
317 
318         /**
319          * Callback
320          * @private
321          * @param idx
322          * @param el
323          */
324         initPanel: function(idx, el) {
325 
326             log('Layout', 'initPanel', arguments);
327 
328             this.addPanel(new ZOOLU.UI.Layout.Panel(el, this.options));
329         },
330 
331         addPanel: function(panel){
332 
333             log('Layout', 'addPanel', arguments);
334 
335             this.panels.push(panel);
336         }
337     };
