  9 (function(window, ZOOLU, $, undefined) {
 10 
 11     'use strict';
 12 
 13     function log() {
 14         if (window.debug) {
 15             window.log.apply(window, $.merge(['ZOOLU', 'UI'], arguments));
 16         }
 17     }
 18 
 19     /** @namespace */
 20     ZOOLU.UI = { };
 21 
 22     /**
 23      * ZOOLU ColumnTree UI Element
 24      *
 25      * @class
 26      * @example
 27      *  var columnTree = new ZOOLU.UI.ColumnTree('#tree', {
 28      *      url: '/nodes',
 29      *      hasChildren: { 'folder': true }
 30      *  });
 31      *
 32      * @borrows ZOOLU.UTIL.Events#fire as #fire
 33      * @borrows ZOOLU.UTIL.Events#on as #on
 34      * @borrows ZOOLU.UTIL.Events#off as #off
 35      * @param {String} container Selector to get the column tree container
 36      * @param {Object} options Default options will be merged with the given options
 37      */
 38     ZOOLU.UI.ColumnTree = function(container, options) {
 39         this.$container = $(container);
 40 
 41         if (!this.$container.length) {
 42             throw "Column container doesn't exist!";
 43         }
 44 
 45         this.$container.addClass('column-tree');
 46 
 47         this.options = $.extend({
 48             hasChildren: { },
 49             urlAddOnForLoadingNodeChildren: '/children'
 50         }, options);
 51 
 52         this.level = 0;
 53         this.selectedId = null;
 54 
 55         this.$selected = null;
 56         this.$currentColumn = null;
 57 
 58         this.data = [];
 59 
 60         log('ColumnTree', 'construct', this);
 61 
 62         ZOOLU.UTIL.Events.enable.call(this);
 63 
 64         this.load();
 65     };
 66 
 67     ZOOLU.UI.ColumnTree.prototype = /** @lends ZOOLU.UI.ColumnTree# */{
 68 
 69         constructor: ZOOLU.UI.ColumnTree,
 70 
 71         /**
 72          * Loads nodes from the given REST service entry point (<code>options.url</code>).
 73          * If there is a selected node, his children will be loaded.
 74          *
 75          * @public
 76          */
 77         load: function() {
 78             log('ColumnTree', 'load');
 79 
 80             if (this.options.url) {
 81                 this.fire('ColumnTree.load');
 82 
 83                 if (this.$container.find('.column').length <= this.level) {
 84                     this.addColumn();
 85                 }
 86 
 87                 this.$currentColumn = $('#column-' + this.level);
 88                 this.$currentColumn.addClass('busy');
 89 
 90                 $.ajax({
 91                     url: this.selectedId ? this.options.url + '/' + this.selectedId + this.options.urlAddOnForLoadingNodeChildren : this.options.url,
 92                     type: 'GET',
 93                     dataType: 'JSON',
 94                     success: function(data, textStatus, jqXHR) {
 95                         log('ColumnTree', 'load', 'success', data);
 96                         this.updateData(data);
 97                         this.updateView();
 98                     }.bind(this),
 99                     error: function(jqXHR, textStatus, errorThrown) {
100                         log('ColumnTree', 'load', 'error', textStatus);
101                         this.$currentColumn.removeClass('busy');
102                         // TODO
103                     }.bind(this)
104                 });
105             }
106         },
107 
108         updateData: function(data) {
109             log('ColumnTree', 'updateData');
110 
111             this.data[this.level] = data;
112         },
113 
114         updateView: function() {
115             log('ColumnTree', 'updateView');
116 
117             if (!this.$currentColumn.length) {
118                 throw "Current column not found!";
119             }
120 
121             this.$currentColumn.removeClass('busy');
122 
123             this.$list = $('<ul class="list"/>');
124 
125             // http://jsperf.com/loops3/2
126             for (var i = -1, length = this.data[this.level].length; ++i < length;) {
127                 this.prepareRow(this.data[this.level][i]).appendTo(this.$list);
128             }
129 
130             this.$currentColumn.html(this.$list);
131 
132             // cleanup
133             delete this.$row;
134             delete this.$list;
135 
136             log('ColumnTree', 'updateView', this.$currentColumn);
137         },
138 
139         addColumn: function() {
140             log('ColumnTree', 'addColumn');
141             this.$container.append($('<div class="column"/>').attr('id', 'column-' + this.level).data('level', this.level));
142         },
143 
144         prepareRow: function(node) {
145             log('ColumnTree', 'prepareRow', arguments);
146 
147             this.$row = $('<li class="row ' + node.type + '"/>').attr('id', 'row-' + node.id + '-' + node.type);
148             this.$row.html(node.name);
149 
150             // store node data to the element
151             this.$row.data('id', node.id);
152             this.$row.data('type', node.type);
153             this.$row.data('level', this.level);
154 
155             this.attachRowObservers(node);
156 
157             return this.$row;
158         },
159 
160         attachRowObservers: function() {
161             log('ColumnTree', 'attachRowObservers', arguments);
162 
163             this.$row.on('click', function(event) {
164                 this.select(event.target);
165             }.bind(this));
166         },
167 
168         select: function(element) {
169             log('ColumnTree', 'select', arguments);
170             this.$selected = $(element);
171             this.fire('ColumnTree.select', [this.$selected]);
172 
173             this.$selected.addClass('selected');
174 
175             this.selectedId = this.$selected.data('id');
176             this.level = this.$selected.data('level');
177 
178             this.$container.find('.column').each(function(index, element) {
179                 log(element);
180                 if ($(element).data('level') > this.level) {
181                     $(element).html('');
182                 }
183             }.bind(this));
184 
185             if (this.options.hasChildren.hasOwnProperty(this.$selected.data('type')) &&
186                 this.options.hasChildren[this.$selected.data('type')] === true) {
187 
188                 this.level++;
189                 this.load();
190             }
191         }
192     };
193 
194 })(window, window.ZOOLU, window.jQuery);
