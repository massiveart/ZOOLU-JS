 10 (function(window, ZOOLU, $, undefined) {
 11 
 12     'use strict';
 13 
 14     function log() {
 15         if (window.debug) {
 16             window.log.apply(window, $.merge(['ZOOLU', 'STORE'], arguments));
 17         }
 18     }
 19 
 20     /** @namespace */
 21     ZOOLU.STORE = { };
 22 
 23     /**
 24      * @namespace
 25      * @example
 26      *  var store = ZOOLU.STORE.Cookie.getInstance();
 27      *  store.set('UI.Layout.height', 80);
 28      *  store.get('UI.Layout.height');
 29      *
 30      * @author <a href="mailto:thomas@chirimoya.at">Thomas Schedler</a>
 31      */
 32     ZOOLU.STORE.Cookie = /** @lends ZOOLU.STORE.Cookie */ (function() {
 33 
 34         var instance;
 35 
 36         function init() {
 37             log('Init ZOOLU.STORE.Cookies');
 38             
 39             var properties = JSON.parse($.cookie('ZOOLU')) || { };
 40 
 41             return {
 42 
 43                 set: function(key, value) {
 44                     properties[key] = value;
 45                     $.cookie('ZOOLU', JSON.stringify(properties));
 46                 },
 47 
 48                 get: function(key) {
 49                     return (properties[key]) ? properties[key] : null;
 50                 }
 51             };
 52         }
 53 
 54         return {
 55             getInstance: function() {
 56                 if (!instance) {
 57                     instance = init();
 58                 }
 59                 return instance;
 60             }
 61         };
 62     })();
 63 
 64 })(window, window.ZOOLU, window.jQuery);
